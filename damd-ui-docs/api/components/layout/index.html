    API: Layout

    Default Layout, a single AppBar with default color and position set to fixed.

    Props

    • title - Title of the AppBar
    • MenuSlot - Left Menu. If not defined the left menu will be automatically hide. See an example with LateralMenu
    • AfterTitleSlot - Space after the title
    • hideThemeSwitch - Hide the ThemeSwitch component
    • children - Positioned bellow the AppBar
    type AppBarProps = React.PropsWithChildren<{
  AfterTitleSlot?: React.FC
  MenuSlot?: React.FC<{ open: boolean }>
  title: string
  hideThemeSwitch?: boolean
}>

    Example

    import {Layout} from 'damd-ui'

...

const App = () => (
  ...
    <Layout
      title='My Application'
      MenuSlot={NavBar}
      hideThemeSwitch
    >
        ...
    </Layout>
  ...
)
